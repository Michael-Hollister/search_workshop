  \begin{itemize}
    
  \item Average branching factor:
    $b$
  \item Depth of goal node: $d$
  \item<1-> BFS
    \begin{itemize}
    \item<2-> Runtime: $O(d^b)$
    \item<3-> Space: $O(d^b)$
    \end{itemize}
    
  \item<4-> DFS
    \begin{itemize}
    \item<5-> Runtime: $O(d^b)$
    \item<6-> Space: $O(d)$
